196         var deltaX = startX - origX, deltaY = startY - origY;
197         document.addEventListener("mousemove", moveHandler, true);

199

204             elem.style.left = (e.clientX - deltaX) + "px";
205             elem.style.top = (e.clientY - deltaY) + "px";
206             e.stopPropagation();
207         }
208
